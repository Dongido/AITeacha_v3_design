// --- REMOVED: JWT is generally NOT used for public meet.jit.si rooms ---
// If you need authentication for meet.jit.si, it's a different JWT mechanism.
// const JITSI_JWT = "eyJhbGciOiJSUzI1NiIs...";

const JitsiMeetingPage = () => {
  const navigate = useNavigate();
  const { meetingId } = useParams<{ meetingId: string }>();
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const jitsiContainerRef = useRef<HTMLDivElement>(null);
  const jitsiApiRef = useRef<any>(null);
  const [chatTranscript, setChatTranscript] = useState<string[]>([]);
  const [agentTriggered, setAgentTriggered] = useState(false);
  const [agentStatus, setAgentStatus] = useState<
    "idle" | "triggering" | "running" | "ended" | "error"
  >("idle");

  const loadJitsiScript = useCallback(() => {
    return new Promise<void>((resolve, reject) => {
      const scriptId = "jitsi-external-api-script";
      if (document.getElementById(scriptId)) {
        resolve(); // Script already loaded
        return;
      }

      const script = document.createElement("script");
      // --- CHANGED: Load external_api.js directly from meet.jit.si ---
      script.src = `https://${JITSI_DOMAIN}/external_api.js`;
      script.id = scriptId;
      script.async = true;
      script.onload = () => resolve();
      script.onerror = () =>
        reject(
          new Error(`Failed to load Jitsi external_api.js from ${JITSI_DOMAIN}`)
        );
      document.body.appendChild(script);
    });
  }, []); // No dependencies needed as JITSI_DOMAIN is constant

  const handleMeetingEnd = useCallback(async () => {
    console.log("Frontend: Meeting ending sequence initiated.");
    setAgentStatus("ended");

    try {
      const response = await fetch(
        `${BACKEND_AGENT_SERVER_URL}/api/meeting-ended`,
        {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ meetingId }),
        }
      );

      if (response.ok) {
        console.log("Frontend: Backend successfully notified of meeting end.");
      } else {
        console.error(
          "Frontend: Failed to notify backend of meeting end:",
          await response.text()
        );
      }
    } catch (error) {
      console.error(
        "Frontend: Network error notifying backend of meeting end:",
        error
      );
    } finally {
      setTimeout(() => {
        if (jitsiApiRef.current) {
          try {
            jitsiApiRef.current.dispose(); // Dispose before navigating
            jitsiApiRef.current = null;
            console.log("Jitsi API disposed during meeting end.");
          } catch (e) {
            console.error("Error disposing Jitsi API:", e);
          }
        }
        navigate(`/dashboard/liveclass/details/${meetingId}`);
      }, 2000);
    }
  }, [meetingId, navigate]);

  const triggerJitsiAgent = useCallback(
    async (meetingUrl: string, agentDisplayName: string) => {
      setAgentStatus("triggering");
      try {
        console.log(
          `Attempting to trigger Jitsi agent via backend for URL: ${meetingUrl}`
        );
        const response = await fetch(
          `${BACKEND_AGENT_SERVER_URL}/start-jitsi-agent`,
          {
            method: "POST",
            headers: {
              "Content-Type": "application/json",
            },
            body: JSON.stringify({
              meetingUrl: meetingUrl,
              displayName: agentDisplayName,
            }),
          }
        );

        const data = await response.json();

        if (response.ok) {
          console.log("Jitsi agent successfully triggered:", data.message);
          setAgentTriggered(true);
          setAgentStatus("running");
        } else {
          console.error(
            "Failed to trigger Jitsi agent:",
            data.error || response.statusText
          );
          setAgentStatus("error");
        }
      } catch (fetchError: any) {
        console.error(
          "Network error when trying to trigger Jitsi agent:",
          fetchError
        );
        setAgentStatus("error");
      }
    },
    []
  );

  useEffect(() => {
    if (!meetingId) {
      setError("Meeting code is missing. Cannot start Jitsi meeting.");
      setLoading(false);
      return;
    }

    setLoading(true);

    const initializeJitsi = async () => {
      try {
        await loadJitsiScript();
      } catch (scriptError: any) {
        console.error("Script loading error:", scriptError);
        setError(scriptError.message || "Failed to load Jitsi resources.");
        setLoading(false);
        return;
      }

      // Check if the container ref is available *after* the script loads
      if (!jitsiContainerRef.current) {
        setError(
          "Jitsi container not found after script load. This is unexpected."
        );
        setLoading(false);
        return;
      }

      // --- CHANGED: Full meeting URL construction for meet.jit.si (no APP_ID) ---
      const fullMeetingUrl = `https://${JITSI_DOMAIN}/${meetingId}`;

      const options = {
        roomName: meetingId,
        width: "100%",
        height: "100%",
        parentNode: jitsiContainerRef.current,

        configOverwrite: {
          disableLocalVideoFlip: true,
          backgroundAlpha: 0.5,
          startWithAudioMuted: false,
          startWithVideoMuted: false,
        },
        interfaceConfigOverwrite: {
          VIDEO_LAYOUT_FIT: "nocrop",
          MOBILE_APP_PROMO: false,
          TILE_VIEW_MAX_COLUMNS: 4,
          TOOLBAR_BUTTONS: [
            "microphone",
            "camera",
            "closedcaptions",
            "desktop",
            "fullscreen",
            "fodeviceselection",
            "hangup",
            "profile",
            "chat",
            "recording",
            "livestreaming",
            "etherpad",
            "sharedvideo",
            "settings",
            "raisehand",
            "videoquality",
            "filmstrip",
            "feedback",
            "stats",
            "shortcuts",
            "tileview",
            "videobackgroundblur",
            "download",
            "help",
            "mute-everyone",
            "security",
          ],
          // HIDE_JITSI_LOGO: true, // often ignored for meet.jit.si
        },
      };

      try {
        // --- CHANGED: First argument is JITSI_DOMAIN (now meet.jit.si) ---
        const api = new JitsiMeetExternalAPI(JITSI_DOMAIN, options);
        jitsiApiRef.current = api;

        console.log("Jitsi API instance created:", api);

        // Update meeting URL in database
        updateLiveClassMeetingUrl(parseInt(meetingId), fullMeetingUrl)
          .then(() => {
            console.log("Meeting URL successfully updated in the database.");
          })
          .catch((apiError: any) => {
            console.error(
              "Failed to update meeting URL in DB:",
              apiError.message
            );
          });

        // Event Listeners
        api.addEventListener("videoConferenceJoined", () => {
          console.log("Video conference joined!");
          setLoading(false); // Meeting loaded, turn off skeleton
          if (fullMeetingUrl && !agentTriggered) {
            triggerJitsiAgent(fullMeetingUrl, `AiTeacha Agent`);
          }
        });

        api.addEventListener("readyToClose", handleMeetingEnd);
        api.addEventListener("hangup", handleMeetingEnd);

        api.addEventListener(
          "incomingMessage",
          (message: {
            from: string;
            message: string;
            privateMessage: boolean;
          }) => {
            const timestamp = new Date().toLocaleTimeString();
            const sender = message.from;
            const text = message.message;
            const isPrivate = message.privateMessage ? "[PRIVATE] " : "";
            const fullMessage = `${timestamp} - ${isPrivate}${sender}: ${text}`;
            console.log("Incoming Chat Message:", fullMessage);
            setChatTranscript((prevTranscript) => [
              ...prevTranscript,
              fullMessage,
            ]);
          }
        );
      } catch (apiInitError: any) {
        console.error("Error initializing JitsiMeetExternalAPI:", apiInitError);
        setError("Failed to initialize Jitsi meeting. Please try again.");
        setLoading(false);
      }
    };

    initializeJitsi();

    // Cleanup function
    return () => {
      if (jitsiApiRef.current) {
        console.log("Disposing of Jitsi API instance on component unmount.");
        try {
          jitsiApiRef.current.dispose();
          jitsiApiRef.current = null;
        } catch (e) {
          console.error("Error disposing Jitsi API during cleanup:", e);
        }
      }
    };
  }, [
    meetingId,
    loadJitsiScript,
    handleMeetingEnd,
    triggerJitsiAgent,
    updateLiveClassMeetingUrl,
    agentTriggered,
  ]);

  return (
    <div className="min-h-screen bg-gray-100 flex flex-col items-center justify-center p-4">
      <h1 className="text-3xl font-bold text-gray-800 mb-6">
        Jitsi Meeting: {meetingId}
      </h1>

      {/* Conditionally render Skeleton/Error or Jitsi container */}
      {loading && !error && (
        <div className="flex flex-col items-center justify-center w-full max-w-4xl h-[600px] rounded-lg bg-white shadow-xl overflow-hidden border border-gray-200">
          <h2 className="text-2xl font-bold text-gray-800 mb-4">
            Loading Meeting...
          </h2>
          <Skeleton className="w-full h-full rounded-lg" />
          <p className="text-gray-600 mt-4">
            Preparing your virtual classroom.
          </p>
        </div>
      )}

      {error && (
        <div className="flex justify-center items-center w-full max-w-4xl h-[600px] bg-red-50 p-4 rounded-lg shadow-md">
          <div className="text-center text-red-700 bg-red-100 p-6 rounded-lg shadow-md">
            <h2 className="text-xl font-semibold mb-2">Error!</h2>
            <p>{error}</p>
            <button
              onClick={() => window.location.reload()}
              className="mt-4 px-6 py-2 bg-red-600 text-white rounded-md hover:bg-red-700 transition-colors"
            >
              Try Again
            </button>
          </div>
        </div>
      )}

      {/* Always render the Jitsi container, but hide it if loading or error */}
      <div
        ref={jitsiContainerRef}
        style={{
          width: "100%",
          height: "600px", // Maintain consistent height
          maxWidth: "4xl",
          backgroundColor: "white",
          borderRadius: "0.5rem", // rounded-lg
          boxShadow:
            "0 20px 25px -5px rgba(0, 0, 0, 0.1), 0 8px 10px -6px rgba(0, 0, 0, 0.1)", // shadow-xl
          overflow: "hidden",
          border: "1px solid #e2e8f0", // border border-gray-200
          display: loading || error ? "none" : "block", // Hide when loading or error
        }}
      />

      <p className="text-gray-600 mt-4 text-center">
        Your meeting is live. Ensure your microphone and camera are enabled.
      </p>

      <div className="mt-4 p-3 rounded-lg text-sm font-medium">
        {agentStatus === "idle" && (
          <p className="text-gray-700">Agent Status: Idle</p>
        )}
        {agentStatus === "triggering" && (
          <p className="text-blue-600">Agent Status: Triggering...</p>
        )}
        {agentStatus === "running" && (
          <p className="text-green-600">Agent Status: Running</p>
        )}
        {agentStatus === "ended" && (
          <p className="text-purple-600">Agent Status: Ended</p>
        )}
        {agentStatus === "error" && (
          <p className="text-red-600">Agent Status: Error starting</p>
        )}
      </div>

      {chatTranscript.length > 0 && (
        <div className="mt-8 w-full max-w-4xl bg-white p-6 rounded-lg shadow-xl border border-gray-200">
          <h2 className="text-xl font-bold text-gray-800 mb-4">
            Chat Transcript
          </h2>
          <div className="h-48 overflow-y-auto border border-gray-300 p-3 rounded-md bg-gray-50">
            {chatTranscript.map((msg, index) => (
              <p key={index} className="text-sm text-gray-700 mb-1">
                {msg}
              </p>
            ))}
          </div>
        </div>
      )}
    </div>
  );
};

export default JitsiMeetingPage;
